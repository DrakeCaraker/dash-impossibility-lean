% trilemma.tex — The Attribution Impossibility trilemma figure
% Compiles standalone:  cd paper/figures && pdflatex trilemma.tex
% Include in paper:     \includegraphics[width=\textwidth]{figures/trilemma.pdf}
%                   or: see figure environment at bottom of this file
%
\documentclass[border=6pt]{standalone}

\usepackage{tikz}
\usepackage{amsmath,amssymb}
\usepackage{xcolor}

\usetikzlibrary{positioning, arrows.meta, calc, decorations.pathreplacing}

\definecolor{familyA}{RGB}{230,120,30}   % orange
\definecolor{familyB}{RGB}{40,100,200}   % blue
\definecolor{impossiblered}{RGB}{200,30,30}
\definecolor{bardark}{RGB}{60,60,60}
\definecolor{barlight}{RGB}{160,160,160}
\definecolor{dashgreen}{RGB}{50,150,80}
\definecolor{trivia}{RGB}{140,140,140}

\begin{document}
\begin{tikzpicture}[
    every node/.style={font=\sffamily},
    >=Stealth
  ]

  % ============================================================
  %  PANEL A — Trilemma Triangle (left, ~60% width)
  % ============================================================
  \begin{scope}[shift={(0,0)}]

    % Equilateral triangle, side length 5.6
    \coordinate (F) at (0, 4.85);      % top: Faithful
    \coordinate (S) at (-2.8, 0);      % bottom-left: Stable
    \coordinate (C) at (2.8, 0);       % bottom-right: Complete

    % Draw edges (thick)
    \draw[line width=1.2pt, black!70] (F) -- (S) -- (C) -- cycle;

    % Vertex labels (bold, large)
    \node[above=6pt of F, font=\sffamily\bfseries\large] {Faithful};
    \node[below left=5pt and 2pt of S, font=\sffamily\bfseries\large] {Stable};
    \node[below right=5pt and 2pt of C, font=\sffamily\bfseries\large] {Complete};

    % Small filled circles at vertices
    \fill (F) circle (3pt);
    \fill (S) circle (3pt);
    \fill (C) circle (3pt);

    % --- Central "Impossible" marker ---
    \coordinate (center) at (0, 1.62);

    % Red circle with X
    \draw[impossiblered, line width=2.5pt] (center) circle (0.55);
    \draw[impossiblered, line width=2.5pt]
      ($(center)+(-0.38,-0.38)$) -- ($(center)+(0.38,0.38)$);
    \draw[impossiblered, line width=2.5pt]
      ($(center)+(-0.38,0.38)$) -- ($(center)+(0.38,-0.38)$);

    % "Impossible" text
    \node[below=10pt of center, font=\sffamily\bfseries,
          text=impossiblered] {Impossible};

    % --- Edge annotations ---

    % Left edge: Faithful <-> Stable — Family B (DASH)
    \node[rotate=59, anchor=south, font=\sffamily\bfseries,
          text=familyB] at ($(F)!0.45!(S) + (-0.35,0.15)$)
      {Family B (DASH)};
    \node[rotate=59, anchor=north, font=\sffamily\small,
          text=familyB!80] at ($(F)!0.55!(S) + (-0.22,-0.05)$)
      {stable + ties};

    % Right edge: Faithful <-> Complete — Family A
    \node[rotate=-59, anchor=south, font=\sffamily\bfseries,
          text=familyA] at ($(F)!0.45!(C) + (0.35,0.15)$)
      {Family A};
    \node[rotate=-59, anchor=north, font=\sffamily\small,
          text=familyA!80] at ($(F)!0.55!(C) + (0.22,-0.05)$)
      {unstable (50\% flips)};

    % Bottom edge: Stable <-> Complete — Drop faithful
    \node[anchor=south, font=\sffamily\bfseries,
          text=trivia] at ($(S)!0.5!(C) + (0,-0.55)$)
      {Drop faithful};
    \node[anchor=north, font=\sffamily\small,
          text=trivia] at ($(S)!0.5!(C) + (0,-0.55)$)
      {trivial};

    % Panel label
    \node[anchor=north west, font=\sffamily\bfseries\large]
      at (-3.6, 5.8) {\textsf{A}};

  \end{scope}

  % ============================================================
  %  PANEL B — Before / After DASH (right, ~40% width)
  % ============================================================
  \begin{scope}[shift={(7.5, 0)}]

    \newcommand{\barwidth}{0.35}
    \newcommand{\gapwidth}{0.18}

    % Panel label
    \node[anchor=north west, font=\sffamily\bfseries\large]
      at (-1.8, 5.8) {\textsf{B}};

    % ----------------------------------------------------------
    %  Top sub-panel: Before DASH (lowered for alignment)
    % ----------------------------------------------------------
    \node[anchor=south, font=\sffamily\bfseries]
      at (0.9, 4.75) {Before DASH};

    % Model 1 (left bar chart)
    \begin{scope}[shift={(-0.5, 2.65)}]
      \node[anchor=south, font=\sffamily\footnotesize, text=black!70]
        at (0.44, 1.95) {Model 1};

      % Feature A: tall
      \fill[bardark] (0, 0) rectangle (\barwidth, 1.7);
      \node[below=2pt, font=\sffamily\scriptsize] at (\barwidth/2, 0) {A};

      % Feature B: short
      \fill[barlight] (\barwidth+\gapwidth, 0)
        rectangle (2*\barwidth+\gapwidth, 0.6);
      \node[below=2pt, font=\sffamily\scriptsize]
        at (\barwidth+\gapwidth+\barwidth/2, 0) {B};

      % Baseline
      \draw[black!40] (-0.1, 0) -- (2*\barwidth+\gapwidth+0.1, 0);
    \end{scope}

    % "retrain" arrow (centered between bar groups)
    \draw[->, line width=1pt, black!60]
      (0.9, 3.8) -- (1.55, 3.8)
      node[midway, above, font=\sffamily\scriptsize, text=black!60] {retrain};

    % Model 2 (right bar chart) — reversed
    \begin{scope}[shift={(1.4, 2.65)}]
      \node[anchor=south, font=\sffamily\footnotesize, text=black!70]
        at (0.44, 1.95) {Model 2};

      % Feature A: short (flipped!)
      \fill[barlight] (0, 0) rectangle (\barwidth, 0.6);
      \node[below=2pt, font=\sffamily\scriptsize] at (\barwidth/2, 0) {A};

      % Feature B: tall (flipped!)
      \fill[bardark] (\barwidth+\gapwidth, 0)
        rectangle (2*\barwidth+\gapwidth, 1.7);
      \node[below=2pt, font=\sffamily\scriptsize]
        at (\barwidth+\gapwidth+\barwidth/2, 0) {B};

      % Baseline
      \draw[black!40] (-0.1, 0) -- (2*\barwidth+\gapwidth+0.1, 0);
    \end{scope}

    % Red annotation: ranking flipped
    \draw[impossiblered, thick, decorate,
          decoration={brace, amplitude=4pt, mirror}]
      (-0.6, 2.5) -- (2.45, 2.5)
      node[midway, below=5pt, font=\sffamily\scriptsize,
           text=impossiblered] {ranking flipped!};

    % ----------------------------------------------------------
    %  Bottom sub-panel: After DASH (M=25) — more space from top
    % ----------------------------------------------------------
    \node[anchor=south, font=\sffamily\bfseries]
      at (0.9, 1.55) {After DASH ($M\!=\!25$)};

    % Single bar chart — tied
    \begin{scope}[shift={(0.2, 0)}]
      % Feature A
      \fill[dashgreen!80] (0, 0) rectangle (\barwidth, 1.1);
      \node[below=2pt, font=\sffamily\scriptsize] at (\barwidth/2, 0) {A};

      % Feature B — same height (tied)
      \fill[dashgreen!80] (\barwidth+\gapwidth, 0)
        rectangle (2*\barwidth+\gapwidth, 1.1);
      \node[below=2pt, font=\sffamily\scriptsize]
        at (\barwidth+\gapwidth+\barwidth/2, 0) {B};

      % Baseline
      \draw[black!40] (-0.1, 0) -- (2*\barwidth+\gapwidth+0.1, 0);

      % Tie bracket
      \draw[dashgreen!70!black, thick, decorate,
            decoration={brace, amplitude=3pt}]
        (0, 1.18) -- (2*\barwidth+\gapwidth, 1.18)
        node[midway, above=4pt, font=\sffamily\footnotesize,
             text=dashgreen!70!black] {tied (honest)};
    \end{scope}

  \end{scope}

\end{tikzpicture}
\end{document}
